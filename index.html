onclick, onchange, oninput, ondblclick

onclick

                        
                            <button onclick="OnClickButton()">onclick</button>
                        
                    
                        
                            function OnClickButton() {
                                console.log("I like cows")
                            }

onchange

                        
                            <input type="range" onchange="OnChangeInput()" id="ID">
                        
                    
                        
                            function OnChangeInput() {
                                console.log(document.getElementById('ID').value)
                            }

oninput

                        
                            <input type="range" onchange="OnInputInput()" id="ID">
                        
                    
                        
                            function OnInputInput() {
                                console.log(document.getElementById('ID').value)
                            }
