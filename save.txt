        </p>
        <pre>
            <code>
                &lt;button onclick=&quot;OnClickButton()&quot;&gt;onclick&lt;/button&gt;
            </code>
        </pre>
        <pre>
            <code>
                function OnClickButton() {
                    console.log(&quot;I like cows&quot;)
                }
            </code>
        </pre>
        <p>
            onchange
        </p>
        <pre>
            <code>
                &lt;input type=&quot;range&quot; onchange=&quot;OnChangeInput()&quot; id=&quot;ID&quot;&gt;
            </code>
        </pre>
        <pre>
            <code>
                function OnChangeInput() {
                    console.log(document.getElementById('ID').value)
                }
            </code>
        </pre>
